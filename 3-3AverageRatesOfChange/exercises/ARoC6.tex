\begin{exercise}

Let $f(x) = \frac{1}{x}$.

\begin{enumerate}
\item Compute $\av_{[1, 2]}$.

$\av_{[1, 2]} = \answer{-\frac{1}{2}}$.  

\item Compute $\av_{[100, 101]}$.

$\av_{[100, 101]} = \answer{-\frac{1}{10100}}$.

	
\end{enumerate}

\end{exercise}
