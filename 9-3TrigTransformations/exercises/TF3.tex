\begin{exercise}
Select all the functions that are equal to the sine function $\sin(x)$. Remember that in order for two functions to be equal, they must have the same domain and range.

\begin{selectAll}
\choice{$\cos(x)$}
\choice{$\cos\left(x + \frac{\pi}{2}\right)$}
\choice[correct]{$\cos\left(x - \frac{\pi}{2}\right)$}
\choice{$\sin(x + \pi)$}
\choice[correct]{$\sin(x + 4\pi)$}
\choice{$\tan(x)\cos(x)$}
\choice{$\cot(x)\sin(x)$}
\choice[correct]{$-\sin(-x)$}
\choice{$\sin(-x)$}

\end{selectAll}


\end{exercise}
