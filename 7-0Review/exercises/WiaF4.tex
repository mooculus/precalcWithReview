\begin{exercise}
Select all of the relations which represent  $y$ as a function of $x$.

\begin{selectAll}
\choice[correct]
{
 $y = x^{3} - x$
}
\choice[correct]
{
$y = \sqrt{x - 2}$
}
\choice[correct]
{
 $x^{3}y = -4$
}\choice
{
$x^{2} - y^{2} = 1$
}\choice[correct]
{
$y = \dfrac{x}{x^{2} - 9}$
}\choice
{
$x = -6$
}\choice
{
$x = y^2 + 4$
}
\choice[correct]
{
$y = x^2 + 4$
}
\choice
{
 $x^2 + y^2 = 4$
}\choice[correct]
{
$y = \sqrt{4-x^2}$
}\choice
{
$x^2 - y^2 = 4$
}\choice[correct]
{
$x^3 + y^3 = 4$
}
\end{selectAll}

\end{exercise}
