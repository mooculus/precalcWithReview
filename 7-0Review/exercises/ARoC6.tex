\begin{exercise}

Let $f(x) = \frac{1}{x^2}$.

\begin{enumerate}
\item Compute $\av_{[2, 4]}$.

$\av_{[2, 4]} = \answer{-\frac{3}{32}}$.  

\item Compute $\av_{[100, 101]}$.

$\av_{[100, 101]} = \answer{-\frac{201}{102010000}}$.

	
\end{enumerate}

\end{exercise}
