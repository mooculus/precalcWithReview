	In each part, an invertible function $f$ will be defined. For each function, find its inverse.

	\begin{exercise}	
		$f(x) = 4x - 7$
		
		\[f^{-1}(x) = \answer{\frac{x+7}{4}}\]
	\end{exercise}
	
	\begin{exercise}	
		$f(x) = \frac{x + 2 }{9} - 1$
		
		\[f^{-1}(x) = \answer{9(x+1)- 2}\]
	\end{exercise}

	\begin{exercise}			
		$f(x) = \sqrt[3]{x+4} - 7$
		
		\[f^{-1}(x) = \answer{(x + 7)^3-4}\]
	\end{exercise}
