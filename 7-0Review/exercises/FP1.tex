Let $f$ be a function defined as follows.

$$
f(x) = \begin{cases}
-x, & x < 0 \\
x^2, & x \ge 0
\end{cases}
$$

\begin{exercise}
\begin{enumerate}
\item Compute $f(1)$.

$f(1) = \answer{1}$ 

\item Compute $f(-1)$.

$f(-1) = \answer{1}$

\item The calculations in parts (a) and (b) above show that $f$ is
\begin{multipleChoice}
\choice{neither even nor odd.}
\choice{even but not odd.}
\choice{odd but not even.}
\choice{both even and odd.}
\choice[correct]{not odd, but $f$ may not be even.}
\choice{not even, but $f$ may not be odd.}
\end{multipleChoice}
\end{enumerate}
\end{exercise}
\begin{exercise}
\begin{enumerate}
\item Compute $f(3)$.

$f(3) = \answer{9}$ 

\item Compute $f(-3)$.

$f(-3) = \answer{3}$

\item The calculations in parts (a) and (b) above show that $f$ is
\begin{multipleChoice}
\choice[correct]{neither even nor odd.}
\choice{even, but not odd.}
\choice{odd, but not even.}
\choice{both even and odd.}
\choice{The calculations do not say anything about whether $f$ is even or odd.}
\end{multipleChoice}
	
\end{enumerate}

\end{exercise}
