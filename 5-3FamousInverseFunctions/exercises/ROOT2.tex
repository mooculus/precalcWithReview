\begin{exercise}
How many solutions are there to $2x^5-1=0$? $\answer{1}$
\begin{exercise}
What is the solution?

$x=\answer{(0.5)^{.2}}$
\end{exercise}
\end{exercise}


\begin{exercise}
How many solutions are there to $3x^4+4=7$? $\answer{2}$
\begin{exercise}
What are the solutions (from smaller to larger)?

$x=\answer{-1}$ and $x=\answer{1}$
\end{exercise}
\end{exercise}
