Use the given pair of functions to find and simplify expressions for the following functions and state the domain of each using interval notation.

\begin{exercise}
For $f(x) = x^2 -x+1$ and $g(x) = 3x-5$ 
\begin{itemize}
\item  $(g \circ f)(x)=\answer{3x^2-3x-2}$ with domain $\left(\answer{-\infty},\answer{\infty}\right)$
\item  $(f \circ g)(x)=\answer{9x^2-33x+31}$ with domain $\left(\answer{-\infty},\answer{\infty}\right)$
\item  $(f \circ f)(x)=\answer{x^4-2x^3+2x^2-x+1}$ with domain $\left(\answer{-\infty},\answer{\infty}\right)$
\end{itemize}
\end{exercise}


\begin{exercise}
For $f(x) = x^2-4$ and $g(x) = |x|$ 
\begin{itemize}
\item  $(g \circ f)(x)=\answer{|x^2-4|}$ with domain $\left(\answer{-\infty},\answer{\infty}\right)$
\item  $(f \circ g)(x)=\answer{x^2-4}$ with domain $\left(\answer{-\infty},\answer{\infty}\right)$
\item  $(f \circ f)(x)=\answer{x^4-8x^2+12}$ with domain $\left(\answer{-\infty},\answer{\infty}\right)$
\end{itemize}
\end{exercise}


\begin{exercise}
For $f(x) = 3x-5$ and $g(x) = \sqrt{x}$ 
\begin{itemize}
\item  $(g \circ f)(x)=\answer{ \sqrt{3x-5}}$ with domain $\left[\answer{\frac{5}{3}},\answer{\infty}\right)$
\item  $(f \circ g)(x)=\answer{3\sqrt{x}-5}$ with domain $\left[\answer{0},\answer{\infty}\right)$
\item  $(f \circ f)(x)=\answer{9x-20}$ with domain $\left(\answer{-\infty},\answer{\infty}\right)$
\end{itemize}
\end{exercise}


\begin{exercise}
For    $f(x) = \frac{x}{2x+1}$ and $g(x) = \frac{2x+1}{x}$
\begin{itemize}
\item  $(g \circ f)(x)=\answer{\frac{4x+1}{x}}$ with domain $\left(\answer{-\infty}, \answer{-\frac{1}{2}}\right) \cup \left(\answer{-\frac{1}{2}}, \answer{0}\right), \cup \left(\answer{0}, \answer{\infty}\right)$
\item  $(f \circ g)(x)=\answer{\frac{2x+1}{5x+2}}$ with domain $\left(\answer{-\infty}, \answer{-\frac{2}{5}}\right) \cup \left(-\answer{\frac{2}{5}},\answer{0}\right) \cup \left(\answer{0},\answer{\infty}\right)$
\item  $(f \circ f)(x)=\answer{\frac{x}{4x+1}}$ with domain $\left(\answer{-\infty}, \answer{-\frac{1}{2}}\right) \cup \left(\answer{-\frac{1}{2}}, \answer{-\frac{1}{4}} \right) \cup \left(\answer{-\frac{1}{4}},\answer{\infty}\right)$
\end{itemize}
\end{exercise}


\begin{exercise}
For   $f(x) = |x|$ and $g(x) = \sqrt{4-x}$
\begin{itemize}
\item  $(g \circ f)(x)=\answer{\sqrt{4-|x|}}$ with domain $\left[\answer{-4},\answer{4}\right]$
\item  $(f \circ g)(x)=\answer{|\sqrt{4-x}|}$ with domain $\left(\answer{-\infty},\answer{4}\right]$
\item  $(f \circ f)(x)=\answer{|x|}$ with domain $\left(\answer{-\infty},\answer{\infty}\right)$
\end{itemize}
\end{exercise}
