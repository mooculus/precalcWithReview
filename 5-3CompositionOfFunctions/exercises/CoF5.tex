\begin{exercise}

Let functions \(r\) and \(s\) be defined by the table below.

$$
\begin{array}{rrrrrrrrrr}
t&-4&-3&-2&-1&0&1&2&3&4\\
\hline
r(t)&4&1&2&3&0&-3&2&-1&-4\\
\hline
s(t)&-5&-6&-7&-8&0&8&7&6&5
\end{array}
$$

\begin{problem}
Determine:
\begin{itemize}
\item \((s \circ r)(3)=\answer{-8}\)
\item \((s \circ r)(-4)=\answer{5}\)
\item \((s \circ r)(0)=\answer{0}\) 
\end{itemize}
\end{problem}
