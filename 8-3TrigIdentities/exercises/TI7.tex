\begin{exercise}
Say $\sin(\theta) = -\frac{7}{25}$, where $\frac{3\pi}{2} < \theta < 2\pi$. Then,

\begin{enumerate}
\item $\sin(2\theta) = \answer{-\frac{336}{625}}$
\item $\cos(2\theta) = \answer{\frac{527}{625}}$
\item $\tan(2\theta) = \answer{-\frac{336}{527}}$
\end{enumerate}
	
\end{exercise}
