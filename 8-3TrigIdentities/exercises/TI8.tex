\begin{exercise}
Say $\cos(\theta) = \frac{28}{53}$, where $0 < \theta < \frac{\pi}{2}$. Then,

\begin{enumerate}
\item $\sin(2\theta) = \answer{\frac{2520}{2809}}$
\item $\cos(2\theta) = \answer{-\frac{1241}{2809}}$
\item $\tan(2\theta) = \answer{-\frac{2520}{1241}}$
\end{enumerate}
	
\end{exercise}
