\begin{exercise}
Select all equalities which are true where defined for all values of $\theta$.
\begin{selectAll}
\choice[correct]{$\cos(\pi - \theta) = \cos(\theta - \pi)$}
\choice{$\sec(\theta) = -\sec(\theta)$}
\choice[correct]{$2 + \tan^2(\theta) = \csc^2(\theta) - \cot^2(\theta) + \sec^2(\theta)$}
\choice[correct]{$\frac{1 - \sin^2(\theta)}{\sin^2(\theta)} = \cot^2(\theta)$}
\choice[correct]{$\cos(2\theta) = \cos^2(\theta) - \sin^2(\theta)$}
\end{selectAll}

\end{exercise}
