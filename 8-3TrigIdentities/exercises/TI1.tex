\begin{exercise}
State whether each equality is true or false where defined for all values of $\theta$.

\begin{enumerate} 
\item $\cos(\pi - \theta) = \cos(\theta - \pi)$
\begin{multipleChoice}  
\choice[correct]{True}
\choice{False}
\end{multipleChoice}  

\item $\sec(\theta) = -\sec(\theta)$
\begin{multipleChoice}  
\choice{True}
\choice[correct]{False}
\end{multipleChoice}  

\item $2 + \tan^2(\theta) = \csc^2(\theta) - \cot^2(\theta) + \sec^2(\theta)$
\begin{multipleChoice}  
\choice[correct]{True}
\choice{False}
\end{multipleChoice}  

\item $\frac{1 - \sin^2(\theta)}{\sin^2(\theta)} = \cot^2(\theta)$
\begin{multipleChoice}  
\choice[correct]{True}
\choice{False}
\end{multipleChoice}  

\item $\cos(2\theta) = \cos^2(\theta) - \sin^2(\theta)$
\begin{multipleChoice}  
\choice[correct]{True}
\choice{False}
\end{multipleChoice}  

\end{enumerate}



\end{exercise}
