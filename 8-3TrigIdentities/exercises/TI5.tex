\begin{exercise}

	Which of the following are equal to \[ \tan(2\theta) \]
	
	Select any/all correct answer/answers.
	\begin{selectAll}
		\choice[correct]{$\displaystyle \frac{2\sin(\theta)\cos(\theta)}{1-2\sin^2(\theta)}$.}
		\choice{$\displaystyle \frac{1-\cos(4\theta)}{1+\cos(4\theta)}$}
		\choice[correct]{$\displaystyle \frac{1}{1-\tan(\theta)}-\frac{1}{1+\tan(\theta)}$.}
		\choice{$\displaystyle 2\tan^2(\theta)-1 $}
	\end{selectAll}
	
\end{exercise}
