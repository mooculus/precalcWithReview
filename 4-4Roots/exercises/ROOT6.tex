\begin{exercise}
How many solutions are there to $x^3=-1$? $\answer{1}$
\begin{exercise}
What is the solution?

$x=\answer{-1}$
\end{exercise}
\end{exercise}

\begin{exercise}

\end{exercise}
\end{exercise}
