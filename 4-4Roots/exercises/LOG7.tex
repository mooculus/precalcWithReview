\begin{exercise}
Which of the following are equivalent to $\ln{e^2}$?\\
\begin{selectAll}
\choice[correct]{$2\ln{e}$}
\choice{$(\ln{e})(\ln{e})$}
\choice[correct]{$\ln{e}+\ln{e}$}
\choice{$2e$}
\choice{$\ln{2e}$}
\choice[correct]{$2$}
\end{selectAll}
\end{exercise}
