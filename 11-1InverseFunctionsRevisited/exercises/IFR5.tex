\begin{exercise}
For each function given below, select whether or not it is invertible.
\begin{enumerate}
\item $f(x) = \sin(x)$ 
\begin{multipleChoice}  
\choice{invertible}  
\choice[correct]{not invertible} 
\end{multipleChoice} 

\item $f(x) = \cos(x)$ 
\begin{multipleChoice}  
\choice{invertible}  
\choice[correct]{not invertible} 
\end{multipleChoice} 

\item $f(x) = \sin(x)$ for $-\frac{\pi}{2} \le x \le \frac{\pi}{2}$ 
\begin{multipleChoice}  
\choice[correct]{invertible}  
\choice{not invertible} 
\end{multipleChoice} 

\item $f(x) = \cos(x)$ for $-\frac{\pi}{2} \le x \le \frac{\pi}{2}$ 
\begin{multipleChoice}  
\choice{invertible}  
\choice[correct]{not invertible} 
\end{multipleChoice} 

\item $f(x) = \cos(x)$ for $0 \le x \le \pi$ 
\begin{multipleChoice}  
\choice[correct]{invertible}  
\choice{not invertible} 
\end{multipleChoice} 

\end{enumerate}
\end{exercise}
