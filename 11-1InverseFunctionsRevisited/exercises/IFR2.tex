\begin{exercise}
For each function given below, select whether or not it is invertible.
\begin{enumerate}
\item $f(x) = x^2$ 
\begin{multipleChoice}  
\choice{invertible}  
\choice[correct]{not invertible} 
\end{multipleChoice} 

\item $f(x) = x^2$ for $x \ge 0$ 
\begin{multipleChoice}  
\choice[correct]{invertible}  
\choice{not invertible} 
\end{multipleChoice} 

\item $f(x) = x^2$ for $x \le 0$ 
\begin{multipleChoice}  
\choice[correct]{invertible}  
\choice{not invertible} 
\end{multipleChoice} 

\item $f(x) = x^2$ for $x > 0$ 
\begin{multipleChoice}  
\choice[correct]{invertible}  
\choice{not invertible} 
\end{multipleChoice} 


\item $f(x) = x^2$ for $x \ge 3$ 
\begin{multipleChoice}  
\choice[correct]{invertible}  
\choice{not invertible} 
\end{multipleChoice} 

\item $f(x) = x^2$ for $x \ge -1$ 
\begin{multipleChoice}  
\choice{invertible}  
\choice[correct]{not invertible} 
\end{multipleChoice} 
\end{enumerate}
\end{exercise}
