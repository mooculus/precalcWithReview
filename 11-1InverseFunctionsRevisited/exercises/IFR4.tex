\begin{exercise}
Select all pairs of functions $f$ and $g$ below such that $g = f^{-1}$.
\begin{selectAll}
\choice{$f(x) = \log_3(x + 2)$ and $g(x) = 3^{x - 2}$}


\choice[correct]{$f(x) = x^2$ for $x \le 0$ and $g(x) = -\sqrt{x}$}

\choice{$f(x) = 2|x|$ and $g(x) = \left|\frac{x}{2}\right|$}


\choice[correct]{$f(x) = e^{9x+1} - 4$ and $g(x) = \frac{\ln(x + 4) - 1}{9}$}

\choice{$f(x) = 5(x - 2)^3 +\frac{1}{2}$ and $g(x) = \sqrt[3]{\frac{x - \frac{1}{2}}{5}} - 2$}

\choice[correct]{$f(x) = x$ and $g(x) = x$}

\end{selectAll}
\end{exercise}
