\begin{exercise}
Say $k$ is an invertible function with $k(-3) = 2$ and $k(5) = -2$. For each function given below, select whether or not it is invertible.
\begin{enumerate}
\item $f(x) = k(x + 2)$ 
\begin{multipleChoice}  
\choice[correct]{invertible}  
\choice{not invertible} 
\end{multipleChoice} 

\item $f(x) = k(x)^2$ 
\begin{multipleChoice}  
\choice{invertible}  
\choice[correct]{not invertible} 
\end{multipleChoice} 

\item $f(x) = 2|k(x)|$ 
\begin{multipleChoice}  
\choice{invertible}  
\choice[correct]{not invertible} 
\end{multipleChoice} 

\item $f(x) = -\frac{1}{2}k(x)$ 
\begin{multipleChoice}  
\choice[correct]{invertible}  
\choice{not invertible} 
\end{multipleChoice} 

\item $f(x) = k(x)$ for $x \ge 3$ 
\begin{multipleChoice}  
\choice[correct]{invertible}  
\choice{not invertible} 
\end{multipleChoice} 

\item $f(x) = k(7x) -4$ 
\begin{multipleChoice}  
\choice[correct]{invertible}  
\choice{not invertible} 
\end{multipleChoice} 
\end{enumerate}
\end{exercise}
