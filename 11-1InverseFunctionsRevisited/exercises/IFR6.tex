\begin{exercise}
Select all pairs of functions $f$ and $g$ below such that $g = f^{-1}$.
\begin{selectAll}
\choice[correct]{$f(x) = x + 3$ and $g(x) = -3 + x$}


\choice{$f(x) = 5x - 7$ and $g(x) = \frac{1}{5}(x + 7)$}

\choice{$f(x) = \frac{1 + \frac{1}{x}}{x}$ and $g(x) = x(x - 1)$}


\choice[correct]{$f(x) = \frac{1}{2}x + 9$ and $g(x) = 2x - 18$}

\choice[correct]{$f(x) = 27x^3$ and $g(x) = \frac{\sqrt[3]{x}}{3}$}

\choice{$f(x) = 4$ and $g(x) = \frac{1}{4}$}

\end{selectAll}
\end{exercise}
