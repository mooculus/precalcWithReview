\begin{exercise}
Consider the following system of equations.
$$
\begin{cases}
\sqrt{x + 1} - y & = 0 \\
x^2 + 4y^2 & = 4
\end{cases}
$$
Select all the solutions of the system above.
\begin{selectAll}
\choice{$(4, 0)$}
\choice{$(-4, 0)$}
\choice[correct]{$(0, 1)$}
\choice{$(-1, 0)$}
\choice{$(1, 0)$}
\choice{$(0, -1)$}
\choice{$(4, -1)$}
\choice{$(-1, 4)$}
\end{selectAll}

\end{exercise}
