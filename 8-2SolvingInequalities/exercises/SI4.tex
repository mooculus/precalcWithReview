For each part, use the given figure to solve the inequality. Assume all important behavior is shown in the figure.
\begin{exercise}
\begin{enumerate}
\item Solve the inequality $x^3 - x > 2x$.
\begin{image}
\begin{tikzpicture}
    \begin{axis}[xmin=-3, xmax=4]
         \addplot[domain=-2.2:3.2, <->]{x^3 - x} node{};
	\addplot[<->, dashed, very thick]{2*x} node{}; 
	\node (g) at (axis cs:-1,1) {$y=x^3 - x$};
	\node (f) at (axis cs:3,4) {$y=2x$};
	\addplot[penColor, mark=*, only marks] coordinates {(0,0)};
	\addplot[penColor, mark=*, only marks] coordinates {(1.732,3.464)};
	\addplot[penColor, mark=*, only marks] coordinates {(-1.732,-3.464)};
	\node(l) at (axis cs:0.5,-1) {$(0,0)$};
	\node(r) at (axis cs:.732,4.464) {$(\sqrt{3},2\sqrt{3})$};
	\node(s) at (axis cs:-1,-4.5) {$(-\sqrt{3},-2\sqrt{3})$};	
    \end{axis}
\end{tikzpicture}
\end{image}

\begin{multipleChoice}
\choice[correct]{$(-\sqrt{3}, 0) \cup (\sqrt{3}, \infty)$}
\choice{$(-\sqrt{3}, 0] \cup [\sqrt{3}, \infty)$}
\choice{$(0, \sqrt{3})$}
\choice{$[0, \sqrt{3}]$}
\end{multipleChoice}

\item Solve the inequality $-x + 1 \le \ln(x)$.
\begin{image}
\begin{tikzpicture}
    \begin{axis}[xmin=-4, xmax=4]
         \addplot[domain=0.0001:5, <->, smooth]{ln(x)} node{};
	\addplot[<->, dashed, very thick]{-x + 1} node{}; 
	\node (g) at (axis cs:3,2) {$y=\ln(x)$};
	\node (f) at (axis cs:-2,5) {$y=-x + 1$};
	\addplot[penColor, mark=*, only marks] coordinates {(1,0)};
	\node(l) at (axis cs:1,-1) {$(1,0)$};
    \end{axis}
\end{tikzpicture}
\end{image}

\begin{multipleChoice}
\choice{$(0, 1]$}
\choice{$(0, 1)$}
\choice{$(1, \infty)$}
\choice[correct]{$[1, \infty)$}
\end{multipleChoice}

\end{enumerate}

\end{exercise}
