\begin{exercise}

Let $f(x) = \frac{1}{x^2}$.

\begin{enumerate}
\item Compute $AV_{[2, 4]}$.

$AV_{[2, 4]} = \answer{-\frac{3}{16}}$.  

\item Compute $AV_{[100, 101]}$.

$AV_{[100, 101]} = \answer{-\frac{201}{102010000}}$.

	
\end{enumerate}

\end{exercise}
