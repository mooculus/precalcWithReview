\begin{exercise}
Which of the following equations or statements are equivalent to $\log_5{7}=x$
  \begin{selectAll}
\choice[correct]{$5$ to what power is $7$?}
\choice {$7^x=5$}
\choice {$7$ to what power is $5$?}
\choice[correct] {$5^x=7$}
\choice {$5^7=x$}
\choice[correct] {$\log{5^x}=\log{7}$}
\choice {$\log_7{5}=x$}
 \end{selectAll}
\end{exercise}
