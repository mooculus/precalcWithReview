\begin{exercise}
Find the zeros of the following function.
\[
z(x)=4x(5x-1)(3x+8)(x-\sqrt{5})(x+\sqrt{5})
\]
Enter the $x$ values from smallest to largest


$$x_1 = \answer{-\frac{8}{3}}$$

$$x_2  = \answer{-\sqrt{5}}$$

$$x_3 = \answer{0}$$

$$x_4  = \answer{\frac{1}{5}}$$

$$x_5 = \answer{\sqrt{5}}$$

\end{exercise}
