\begin{exercise}
For each function, select all zeros of the given function.
\begin{enumerate}
\item Let $f$ be a function defined by $f(x) = 3x - 5$.
Select all zeros of $f$.
\begin{selectAll}
\choice{$\frac{1}{3}$}
\choice{$\frac{3}{5}$}
\choice{$1$}
\choice[correct]{$\frac{5}{3}$}
\end{selectAll}

\item Let $g$ be a function defined by $g(x) = 5 - x$.
Select all zeros of $g$.
\begin{selectAll}
\choice{$1$}
\choice{$4$}
\choice[correct]{$5$}
\choice{$-5$}
\end{selectAll}

\item Let $h$ be a function defined by $h(x) = \frac{2 - x}{3}$.
Select all zeros of $h$.
\begin{selectAll}
\choice{$\frac{2}{3}$}
\choice{$\frac{3}{2}$}
\choice{$3$}
\choice[correct]{$2$}
\end{selectAll}

\end{enumerate}

\end{exercise}
