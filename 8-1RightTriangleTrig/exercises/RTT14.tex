\begin{exercise}
  If $\theta$ is the acute angle for which $\tan\theta=\sqrt{2}$, then the value of the sum $\cos\theta+\sec\theta$ is:
  \begin{multipleChoice}
    \choice{$1+\sqrt{2}$}
    \choice{$1+\sqrt{3}$}
    \choice{$\frac{3\sqrt{2}}{2}$}
    \choice[correct]{$\frac{4\sqrt{3}}{3}$}
    \choice{$\frac{5\sqrt{6}}{6}$}
  \end{multipleChoice}
\end{exercise}
