\begin{exercise}

 Consider the graph of a function $f(x) = ab^x+c$.

%\begin{image}
 \begin{tikzpicture}
   \begin{axis}[clip=true,domain=-7:7,xmin=-7,xmax=7,ymin=-7,ymax=7]%\addplot {2*exp(-3*x)-1};
     \addplot {1.2 * .5^x -1};
   \end{axis}
 \end{tikzpicture}
 %\end{image}

 From the graph, answer:

 \begin{enumerate}
 \item The value of $a$ is:
   \begin{multipleChoice}
     \choice[correct]{Positive}
     \choice{Negative}
   \end{multipleChoice}
 \item The value of $b$ is:
   \begin{multipleChoice}
     \choice{Greater than one}
     \choice[correct]{Between zero and one}
   \end{multipleChoice}
 \item  The value of $c$ is:
   \begin{multipleChoice}
     \choice{Positive}
     \choice[correct]{Negative}
   \end{multipleChoice}
 \end{enumerate}

\end{exercise}
