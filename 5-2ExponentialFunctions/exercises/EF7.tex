\begin{exercise}

 Consider the graph of a function $f(x) = ab^x+c$. 

 \begin{image}
   \begin{tikzpicture}
     \begin{axis}
       \addplot[smooth,thick,domain=-7:2, color=penColor]{-2* exp(2*x)+1};
     \end{axis}
   \end{tikzpicture}
 \end{image}

 From the graph, answer:

 \begin{enumerate}
 \item The value of $a$ is:
   \begin{multipleChoice}
     \choice{Positive}
     \choice[correct]{Negative}
   \end{multipleChoice}
 \item The value of $b$ is:
   \begin{multipleChoice}
     \choice[correct]{Greater than one}
     \choice{Between zero and one}
   \end{multipleChoice}
 \item  The value of $c$ is:
   \begin{multipleChoice}
     \choice[correct]{Positive}
     \choice{Negative}
   \end{multipleChoice}
 \end{enumerate}

\end{exercise}
