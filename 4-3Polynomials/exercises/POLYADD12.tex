\begin{exercise}
Select all of the following expressions that are polynomials. 

\begin{selectAll}
\choice[correct]{$x^2 - \frac{2}{19}$}
\choice[correct]{$x^2$}
\choice{$\frac{1}{x}$}
\choice[correct]{$\frac{x^3}{5}$}
\choice[correct]{$\frac{x^3 + 2x - 678}{5}$}
\choice[correct]{$17$}
\choice[correct]{$38x$}
\choice{$\frac{(x - 2)(x + 5)}{x + 2}$}
\choice[correct]{$(x - 2)(x + 5)$}
\choice[correct]{$x^2(x -\pi)^5(x - 1)$}
\choice{$x^2 + x + x^\pi$}
\choice[correct]{$\frac{22}{7}$}
\choice{$2^x + x^2$}
\choice{$\sin(x) + \cos(x)$}
\choice{$|x| + x$}
\choice[correct]{$(x^2 - 25)(x^3 - 27)$}
\choice[correct]{$x + 1$}
\choice{$\frac{2x^2 + 54x - 1}{x}$}
\choice[correct]{$\pi$}
\choice[correct]{$2x^3 + \pi x - \frac{\pi}{4}$}
\choice[correct]{$\frac{1}{5}x^{89} - \frac{2}{9}x^{32} + \frac{2}{\pi}$}
\choice[correct]{$2^5 - 4^9$} 
\choice{$2^{5x} - 4^9$}
\choice{$x^{1/2}$}
\choice[correct]{$x$}
\choice[correct]{$x - x$}

\end{selectAll}
\end{exercise}
