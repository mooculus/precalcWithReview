\begin{exercise}

An object is moving along a horizontal line.  Its position in meters is given by
$$
s(t) = t^2 - 2
$$
where $t$ is in seconds.

Consider the points on the line below.
\begin{image}%[0.75\linewidth]
\begin{tikzpicture}
    \begin{axis}[
        xmin=-4.3,xmax=4.3,ymin=-.1,ymax=.1,
        clip=false,
        unit vector ratio*=1 1 1,
        axis lines=center,
        hide obscured x ticks=false,
        grid = major,
        xtick={-4,-3,...,4},
        hide y axis,
        xlabel=$s$,
        every axis x label/.style={anchor=south}
        ]
        \addplot[only marks,very thick,penColor,mark=*]
        	coordinates{(-2,0) (-1,0) (0,0) (2,0) (4,0)};
	
		\node at (axis cs:-2,0) [penColor,above] {$A$};
		\node at (axis cs:-1,0) [penColor,above] {$B$};
		\node at (axis cs:0,0) [penColor,above] {$C$};
		\node at (axis cs:2,0) [penColor,above] {$D$};
		\node at (axis cs:4,0) [penColor,above] {$E$};
    \end{axis}`
\end{tikzpicture}
\end{image}

The point that corresponds to the position of the particle at $t=1$ is $\answer{B}$.

\begin{exercise}

The average velocity of the object on the interval $[1,3]$. That is, calculate the average rate of change,  $AV_{[1, 3]}$.
$$
AV_{[1, 3]} = \answer{4}\text{m/s.}
$$


\begin{exercise}



The graph of $y=s(t)$ is given below.
\begin{image}%[0.25\linewidth]
	\begin{tikzpicture}
	    \begin{axis}[
	        xmin=-0.3,xmax=3.3,ymin=-2.3,ymax=7.3,
	        clip=true,
	        unit vector ratio*=1 1 1,
	        axis lines=center,
	        hide obscured x ticks=true,
	        grid = major,
	        xtick={0,1,...,3},
	        ytick={-2,-1,...,7},
	        xlabel=$s$,
	        ylabel=$t$,
	        every axis y label/.style={at=(current axis.above origin),anchor=south},
	        every axis x label/.style={at=(current axis.right of origin),anchor=west},
	        ]
	        \addplot[very thick,penColor,domain=0:3.3,samples=50] plot{x^2-2};
	       	\node at (axis cs:1.2,5.1) [penColor,above] {$s=s(t)$};
	    \end{axis}`
	\end{tikzpicture}
\end{image}

Assume that $P$ and $A$ are points on the graph of $s(t)$.  Then
\[
P = \left(1,\answer{-1}\right) \text{ and } A = \left(3,\answer{7}\right).
\]

		\node at (axis cs:1.2,5.1) [penColor2,above] {$y=s(t)$};

The slope of the secant line containing the points $P$ and $A$ is
$$
m_{\text{sec}} = \answer{4}.
$$

What is the connection between the slope of this secant line and the average velocity of the object over the interval $[1,3]$?

\begin{multipleChoice}
	\choice{$m_{\text{sec}}$ is greater than $AV_{[1, 3]}$.}
	\choice[correct]{$m_{\text{sec}}$ equals $AV_{[1, 3]}$.}
	\choice{$m_{\text{sec}}$ is less than $AV_{[1, 3]}$.}
	\choice{There is no connection between $m_{\text{sec}}$ and $AV_{[1, 3]}$.}
\end{multipleChoice}
