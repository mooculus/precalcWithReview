\begin{exercise}
Factor the polynomial into a product of linear factors.\\
\[
f(x) = 3x^2+14x+8
\]
Enter the factor with the larger $x$ value first.\\
$f(x) = (\answer{3}x+\answer{2})(\answer{1}x+\answer{4})$
\end{exercise}
