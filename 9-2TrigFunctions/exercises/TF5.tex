\begin{exercise}
In this exercise, we will find properties of the function $h$ defined by $h(x) = \frac{1}{4}\cos(4x) - 3$.

\begin{enumerate}
\item The domain of $h$ is 
\begin{multipleChoice}
\choice{$(-\infty, 4) \cup (4, \infty)$.}
\choice{$\left[-\frac{13}{4}, -\frac{11}{4}\right]$.}
\choice[correct]{$(- \infty, \infty)$.}
\choice{$(-\frac{13}{4}, -\frac{11}{4})$.}
\choice{$(-\infty, 4] \cup [4, \infty)$.}
\end{multipleChoice}

\item The range of $h$ is 
\begin{multipleChoice}
\choice{$(-\infty, 4) \cup (4, \infty)$}
\choice[correct]{$\left[-\frac{13}{4}, -\frac{11}{4}\right]$}
\choice{$(- \infty, \infty)$}
\choice{$(-\frac{13}{4}, -\frac{11}{4})$}
\choice{$(-\infty, 4] \cup [4, \infty)$}
\end{multipleChoice}

\item The period of $h$ is $\answer{\frac{\pi}{2}}$.

\item The function $h$ is 
\begin{multipleChoice}
\choice{odd.}
\choice[correct]{even.}
\choice{odd and even.}
\choice{neither odd nor even.}
\end{multipleChoice}


\item The amplitude of $h$ is $\answer{\frac{1}{4}}$.
\end{enumerate}



\end{exercise}
