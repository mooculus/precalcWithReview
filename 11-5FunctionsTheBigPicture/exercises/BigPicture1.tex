\begin{exercise}
%
Select all of the relations which represent  $y$ as a function of $x$.
\begin{selectAll}
\choice[correct]{\{$(-3, 7)$, $\;(-2, 7)$, $\;(-1, 1)$, $\;(0, 0)$, $\;(1, 1)$, $\;(2, 7)$, $\;(3, 9)\}$ }
\choice{$\left\{ (-2,1), (-1,6), (-3, -3), (4,1), (-6,5), (4, -1), (6,-5) \right\}$}
\choice[correct]{$y = \sqrt{3x - 4}+1$}
\choice[correct]{$x^{3}y = -2x+3$}
\choice{$x^{2} - y^{2} = 4$}
\choice[correct]{$y = \dfrac{3x+2}{x^{2} - 9}$}
\choice[correct]{$\left\{ (-2,1), (-7,1), (3,4), (4,3), (2,1), (1,0) \right\}$}
\choice[correct]{$\left\{ (1,2), (4,4), (9,6), (16,8), (25,10), (36, 12), \ldots \right\}$}
\choice{\{($x, y) \, | \, x$ is an even integer, and $y$ is an odd integer\}}
\choice[correct]{\{$(x, 1) \, | \, x$ is an irrational number\}}
\choice[correct]{\{$(1, 0)$, $\;(2, 1)$, $\;(4, 2)$, $\;(8, 3)$, $\;(16, 4)$, $\;(32, 5), \;$ \ldots\}}
\choice{$x = -1$}
\choice{$x = y^2 + 9$}
\choice[correct]{$y = x^2 + 9$}
\choice{$x^2 + y^2 = 5$}
\choice[correct]{\{$\ldots, \; (-3, 9)$, $\;(-2, 4)$, $\;(-1, 1)$, $\;(0, 0)$, $\;(1, 1)$, $\;(2, 4)$, $\;(3, 9), \;$ \ldots\}}
\choice[correct]{$y = \frac{1}{7}x^{3} - 2x+4$}
\choice[correct]{\{$(x, \sin(x)) \, | \, x$ is a real number\}}
\choice{\{$(\sin(x), x) \, | \, x$ is a real number\}}
\choice{$\{ (2, y) \, | \, -4 < y < 3\}$}
\choice[correct]{$\{ (x,4) \, | \,  -5 \leq x < 6\}$}
\choice[correct]{$y = \sqrt{5-x^2}$}
\choice{$x^2 - y^2 = 3$}
\choice[correct]{$x^3 + y^3 = 3$}
\choice[correct]{$\{ \left(x,x^4\right) \, | \, \text{$x$ is a real number} \}$}
\choice{$\{ \left(x^4,x\right) \, | \, \text{$x$ is a real number} \}$}
\end{selectAll}
\end{exercise}
