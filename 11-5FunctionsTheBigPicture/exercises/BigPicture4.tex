\begin{exercise}
For each of the following graphs, select the appropriate formula.
\begin{enumerate}
\item The following is the graph of $f(x)$.
\begin{center}
\begin{tikzpicture}
			\begin{axis}[
				samples=100, xmin=-1.5, xmax=7.5, ymin=-2.5,ymax=4.1,    
				axis lines =middle, 
				every axis y label/.style={at=(current axis.above origin),anchor=south},
				every axis x label/.style={at=(current axis.right of origin),anchor=west},
				xtick={-1,...,7}, ytick={-2,...,4},
				grid=major, width=4in,
		                grid style={dashed, gridColor},
				]
				\addplot[color=red, very thick, smooth, domain=-4:7]{log2(x-1)} node[pos=0.6, above left]{\Large{$y=f(x)$}};
			\end{axis}
		\end{tikzpicture}
\end{center}
	Which of the following could be a formula for $f(x)$?	
	\begin{multipleChoice}
		\choice{$2^x-1$}
		\choice{$2^{x-1}$}
		\choice{$(x-1)^2$}
		\choice[correct]{$\log_2(x-1)$}
	\end{multipleChoice}
%
\item The following is the graph of $g(x)$.
\begin{center}
\begin{tikzpicture}
			\begin{axis}[
				xmin=-2.5, xmax=6.5, ymin=-2.5,ymax=4.1,    
				axis lines =middle, 
				every axis y label/.style={at=(current axis.above origin),anchor=south},
				every axis x label/.style={at=(current axis.right of origin),anchor=west},
				xtick={-5,...,5,6}, ytick={-2,...,4},
				grid=major, width=4in,
		                grid style={dashed, gridColor},
				]
				\addplot[color=red, very thick, smooth, domain=-5:6]{e^(x-4)} node[pos=0.6, above left]{\Large{$y=g(x)$}};
			\end{axis}
		\end{tikzpicture}
\end{center}
	Which of the following could be a formula for $g(x)$?	
	\begin{multipleChoice}
		\choice{$e^x-4$}
		\choice[correct]{$e^{x-4}$}
		\choice{$(x+4)^2$}
		\choice{$(x-4)^2$}
		\choice{$\ln(x+4)$}
	\end{multipleChoice}

\item The following is the graph of $h(x)$.
\begin{center}
\begin{tikzpicture}
			\begin{axis}[
				xmin=-5.5, xmax=3.5, ymin=-1.5,ymax=5.1,    
				axis lines =middle, 
				every axis y label/.style={at=(current axis.above origin),anchor=south},
				every axis x label/.style={at=(current axis.right of origin),anchor=west},
				xtick={-5,...,5,6}, ytick={-2,...,4},
				grid=major, width=4in,
		                grid style={dashed, gridColor},
				]
				\addplot[color=red, very thick, smooth, domain=-5:6]{(x+2)^2-1} node[pos=.19, above right]{\Large{$y=h(x)$}};
			\end{axis}
		\end{tikzpicture}
\end{center}
	Which of the following could be a formula for $h(x)$?	
	\begin{multipleChoice}
		\choice{$-2^{x+2}-1$}
		\choice{$-2^{x-2}-1$}
		\choice{$(x-2)^2-1$}
		\choice[correct]{$(x+2)^2-1$}
		\choice{$-\ln(x+2)+1$}
	\end{multipleChoice}
%
\item The following is the graph of $k(x)$.
\begin{center}
\begin{tikzpicture}
			\begin{axis}[
				samples=250, xmin=-3.5, xmax=4.5, ymin=-2.5,ymax=4.1,    
				axis lines =middle, 
				every axis y label/.style={at=(current axis.above origin),anchor=south},
				every axis x label/.style={at=(current axis.right of origin),anchor=west},
				xtick={-5,...,5,6}, ytick={-2,...,4},
				grid=major, width=4in,
		                grid style={dashed, gridColor},
				]
				\addplot[color=red, very thick, smooth, domain=-3:5]{sqrt(x+3)-2} node[pos=.8, above left]{\Large{$y=k(x)$}};
			\end{axis}
		\end{tikzpicture}
\end{center}
	Which of the following could be a formula for $h(x)$?	
	\begin{multipleChoice}
		\choice{$\sqrt{x+2}-3$}
		\choice[correct]{$\sqrt{x+3}-2$}
		\choice{$(x+2)^2-3$}
		\choice{$(x+3)^2-2$}
		\choice{$\ln(x+3)-2$}
		\choice{$\ln(x+2)-3$}
	\end{multipleChoice}
%
\item The following is the graph of $f(x)$.
\begin{image}
\includegraphics[width=1\linewidth]{3cos4x-1Graph.png}
\end{image}
	Which of the following could be a formula for $f(x)$?	
	\begin{multipleChoice}
		\choice{$3\sin(4x)-1$}
		\choice[correct]{$3\cos(4x)-1$}
		\choice{$4\sin(3x)$}
		\choice{$4\cos(3x)$}
	\end{multipleChoice}
%
\item The following is the graph of $g(x)$.
\begin{image}
\includegraphics[width=1\linewidth]{thirdSine5x-Graph.png}
\end{image}
	$g(x) = \answer{\frac{1}{3}}\sin\!\big(\answer{5}x - \frac{\pi}{2}\big)+\answer{\frac{1}{2}}$

\item The following is the graph of $h(x)$.
\begin{image}
\includegraphics[width=1\linewidth]{tanGraphwithHintpoint.png}
\end{image}
What is the parent function of this graph?\\
\wordChoice{\choice{log}\choice{\text{cosine}}\choice[correct]{\text{tangent}}\choice{e}\choice{\text{secant}}\choice{\text{arctangent}}}
\begin{exercise}
	$h(x) = \frac{1}{4}\tan\!\big(\answer{4}x\big)+\answer{-1}$
\end{exercise}
	
\end{enumerate}
\end{exercise}
