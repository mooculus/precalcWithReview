\begin{exercise}

  If $y = 4e^{7x}-8$, solve for $x$ in terms of $y$.

  Answer: $x = \answer{\frac{1}{7}} \ln \left(\answer{{\frac{y+8}{4}}}\right)$.

\end{exercise}
