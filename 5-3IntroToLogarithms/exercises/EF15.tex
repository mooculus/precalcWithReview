\begin{exercise}

  If $y = 10e^{3x}+5$, solve for $x$ in terms of $y$.

  Answer: $x = \answer{\frac{1}{3}} \ln \left(\answer{\frac{y-5}{10}}\right)$.

\end{exercise}
