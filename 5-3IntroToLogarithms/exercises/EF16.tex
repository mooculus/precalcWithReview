\begin{exercise}

  If $y = -6e^{5x}+3$, solve for $x$ in terms of $y$.

  Answer: $x = \answer{\frac{1}{5}} \ln \left(\answer{\frac{3-y}{6}}\right)$.

\end{exercise}
