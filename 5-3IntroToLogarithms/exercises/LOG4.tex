\begin{exercise}
The following is the graph of a logarithmic function, $f(x)$.
\begin{center}
\begin{tikzpicture}
			\begin{axis}[
				xmin=-2.5, xmax=6.5, ymin=-4.5,ymax=2.1,    
				axis lines =middle, 
				every axis y label/.style={at=(current axis.above origin),anchor=south},
				every axis x label/.style={at=(current axis.right of origin),anchor=west},
				xtick={-5,...,6}, ytick={-4,...,2},
				grid=major, width=4in,
		                grid style={dashed, gridColor},
				]
				\addplot[color=red, very thick, smooth, domain=-1.95:8]{-ln(x+2)} node[pos=0.02, above right]{\Large{$y=f(x)$}};
			\end{axis}
		\end{tikzpicture}
\end{center}
	Which of the following could be a formula for $f(x)$?	
	\begin{multipleChoice}
		\choice{$\ln(x+2)$}
		\choice[correct]{$-\ln(x+2)$}
		\choice{$\ln(x-2)$}
		\choice{$-\ln(x-2)$}
		\choice{$e^{x}-2$}
		\choice{$-e^{x}-2$}
	\end{multipleChoice}
\end{exercise}
