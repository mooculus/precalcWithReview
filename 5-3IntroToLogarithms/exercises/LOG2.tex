\begin{exercise}
Which of the following equations or statements are equivalent to $log_3{2}=x$
  \begin{selectAll}
\choice {$2^x=3$}
\choice {$3^2=x$}
\choice[correct]{$3$ to what power is $2$?}
\choice {$2$ to what power is $3$?}
\choice[correct] {$3^x=2$}
\choice[correct] {$\log{3^x}=\log{2}$}
\choice {$log_2{3}=x$}
 \end{selectAll}
\end{exercise}
