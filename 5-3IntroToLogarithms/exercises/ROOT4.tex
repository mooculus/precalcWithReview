\begin{exercise}
How many solutions are there to $x^3=0$? $\answer{1}$
\begin{exercise}
What is the solution?

$x=\answer{0}$
\end{exercise}
\end{exercise}

\begin{exercise}
How many solutions are there to $x^3-9=0$? $\answer{1}$
\begin{exercise}
What is the solution?

$x=\answer{\sqrt[3]{9}}$
\end{exercise}
\end{exercise}

\begin{exercise}

\end{exercise}
\end{exercise}
