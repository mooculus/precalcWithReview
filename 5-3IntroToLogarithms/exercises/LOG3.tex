\begin{exercise}
The following is the graph of a logarithmic function, $f(x)$.
\begin{center}
\begin{tikzpicture}
			\begin{axis}[
				xmin=-4.5, xmax=4.5, ymin=-2.5,ymax=4.1,    
				axis lines =middle, 
				every axis y label/.style={at=(current axis.above origin),anchor=south},
				every axis x label/.style={at=(current axis.right of origin),anchor=west},
				xtick={-5,...,5}, ytick={-2,...,4},
				grid=major, width=4in,
		                grid style={dashed, gridColor},
				]
				\addplot[color=red, very thick, smooth, domain=-5:5]{log2(x+3)} node[pos=0.02, above right]{\Large{$y=f(x)$}};
			\end{axis}
		\end{tikzpicture}
\end{center}
	Which of the following could be a formula for $f(x)$?	
	\begin{multipleChoice}
		\choice[correct]{$\log_2(x+3)$}
		\choice{$log_2(x-3)$}
		\choice{$2^x-3$}
		\choice{$-2^x-3$}
	\end{multipleChoice}
\end{exercise}
