\begin{exercise}
Evaluate $f(3)$ and $f(5)$.\\
$f(x)= x^3+x$\\
$f(3)=\answer{30}$\\
$f(5)=\answer{130}$
\end{exercise}
\begin{exercise}
Find $f(5)-f(3)$.\\
$f(5)-f(3)=\answer{100}$
\end{exercise}
\begin{exercise}
Find $\av_{[3, 5]} f(x)$.\\
$\av_{[3, 5]} f(x) = \answer{50}$
\end{exercise}
\begin{exercise}
Find the slope of the line with the points $(3, f(3))$ and $(5, f(5)$.\\
$m=\answer{50}$
\end{exercise}
