\begin{exercise}
Select all functions that are one-to-one on the given domain.
%
%throw in one or two trig functions vs restricted trig functions
\begin{selectAll}
\choice{$f(x) = x^2 +2$ on $(-4, 8)$}
\choice[correct]{$g(y) = y^2+4$ on $(-8,-4)$}
\choice{$h(w) = 2|w| - 1$ on $\left(-\frac{1}{2},\frac{1}{2}\right)$}
\choice[correct]{$k(v) = 7^{v-4}$ on $(-\infty, \infty)$}
\choice[correct]{$f(z) = 2(z-3)^2+7$ on $(3, \infty)$}
\choice[correct]{$g(\alpha) = 2(\alpha+3)^2+7$ on $(-\infty, -3)$}
\choice{$h(\beta) = 2(\beta-3)^2+7$ on $(-3, \infty)$}
\choice{$k(x) = 2(x+3)^2+7$ on $(-\infty, 3)$}
\choice[correct]{$f(\gamma) = \sin(2\gamma)$ on $\left[-\frac{\pi}{4}, \frac{\pi}{4} \right]$}
\choice{$g(\theta) = \sin(2\theta)$ on $\left[-\frac{\pi}{2}, \frac{\pi}{2} \right]$}
\choice[correct]{$h(w) = \frac{\ln(w+5) - 2}{8}$ on $(-5,\infty)$}
\choice{$k(z) = \frac{\ln(z-2) + 1}{5}$ on $(0,\infty)$}
\choice{$f(\alpha)= \tan\left(\frac{\alpha}{4}\right)$ on $\left(-\frac{\pi}{8},\frac{\pi}{8}\right)$.}
\choice[correct]{$g(\beta)= \tan\left(\frac{\beta}{4}\right)$ on $(-2\pi,2\pi)$.}
\end{selectAll}

\begin{exercise}
Since we established the functions are one-to-one on the given domain, find their inverses:
\begin{enumerate}

\item $g(y) = y^2+4$ on $(-8,-4)$.\\
$g^{-1}(y) = \answer{-\sqrt{y-4}}$

\item $k(v) = 7^{v-4}$ on $(-\infty, \infty)$.\\
$k^{-1}(v) = \answer{\log_7v +4}$

\item $f(z) = 2(z-3)^2+7$ on $(3, \infty)$.\\
$f^{-1}(z) = \answer{\sqrt{\frac{z-7}{2}}+3}$

\item $g(\alpha) = 2(\alpha+3)^2+7$ on $(-\infty, -3)$.\\
$g^{-1}(\alpha) = \answer{-\sqrt{\frac{\alpha-7}{2}}-3}$

\item $f(\gamma) = \sin(2\gamma)$ on $\left[-\frac{\pi}{4}, \frac{\pi}{4} \right]$. \\
$f^{-1}(\gamma) = \answer{2\arcsin(\gamma)}$

\item $h(w) = \frac{\ln(w+5) - 2}{8}$ on $(-5,\infty)$. \\
$h^{-1}(w) = \answer{e^{8w+2}-5}$

\item $g(\beta)= \tan\left(\frac{\beta}{4}\right)$ on $(-2\pi,2\pi)$.\\
$g^{-1}(\beta) = \answer{4\arctan(\beta)}$

\end{enumerate}
\end{exercise}



\end{exercise}
