\begin{exercise}
Consider the following system of equations.
$$
\begin{cases}
x^2 - y^2 & = 9 \\
x^2 + y^2 & = 9
\end{cases}
$$
Select all the solutions of the system above.
\begin{selectAll}
\choice{$(0, 3)$}
\choice{$(0, -3)$}
\choice[correct]{$(3, 0)$}
\choice[correct]{$(-3, 0)$}
\choice{$(3, 3)$}
\choice{$(-3, -3)$}
\choice{$(-3, 3)$}
\choice{$(3, -3)$}
\choice{$(0, 0)$}
\end{selectAll}

\end{exercise}
