For each equation below, determine the type of conic section that most closely corresponds to its graph.
\begin{exercise}
\begin{enumerate}
	\item $\frac{x^2}{25} - 1 = y^2$
\begin{multipleChoice}  
\choice{circle}  
\choice{parabola}  
\choice{ellipse}
\choice[correct]{hyperbola}
\end{multipleChoice}  

	\item $\frac{y^2}{64} + \frac{x^2}{16} = 1$
\begin{multipleChoice}  
\choice{circle}  
\choice{parabola}  
\choice[correct]{ellipse}
\choice{hyperbola}
\end{multipleChoice}  

	\item $\frac{y^2}{20} + \frac{x^2}{20} = 1$
\begin{multipleChoice}  
\choice[correct]{circle}  
\choice{parabola}  
\choice{ellipse}
\choice{hyperbola}
\end{multipleChoice}  

	\item $y^2 = \frac{x}{8}$
\begin{multipleChoice}  
\choice{circle}  
\choice[correct]{parabola}  
\choice{ellipse}
\choice{hyperbola}
\end{multipleChoice}  

\end{enumerate}





\end{exercise}
