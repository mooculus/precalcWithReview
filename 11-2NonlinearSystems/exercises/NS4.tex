\begin{exercise}
Consider the following system of equations.
$$
\begin{cases}
x^2 - y & = 25 \\
x^2 + y^2 & = 25
\end{cases}
$$
Select all the solutions of the system above.
\begin{selectAll}
\choice[correct]{$(-5, 0)$}
\choice[correct]{$(5, 0)$}
\choice{$(0, 5)$}
\choice{$(0, -5)$}
\choice{$(2\sqrt{6}, 0)$}
\choice{$(-2\sqrt{6}, 0)$}
\choice{$(2\sqrt{6}, 1)$}
\choice{$(-2\sqrt{6}, 1)$}
\choice[correct]{$(2\sqrt{6}, -1)$}
\choice[correct]{$(-2\sqrt{6}, -1)$}
\choice{$(1, 2\sqrt{6})$}
\choice{$(1, -2\sqrt{6})$}
\choice{$(-1, 2\sqrt{6})$}
\choice{$(-1, -2\sqrt{6})$}
\end{selectAll}

\end{exercise}
