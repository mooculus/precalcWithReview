\begin{exercise}
Consider the following system of equations.
$$
\begin{cases}
x^2 +y^2 & = 16 \\
16x^2 + 4y^2 & = 64
\end{cases}
$$
Select all the solutions of the system above.
\begin{selectAll}
\choice[correct]{$(0, 4)$}
\choice[correct]{$(0, -4)$}
\choice{$(4, 0)$}
\choice{$(-4, 0)$}
\choice{$(4, 4)$}
\choice{$(-4, -4)$}
\choice{$(-4, 4)$}
\choice{$(4, -4)$}
\choice{$(0, 0)$}
\end{selectAll}

\end{exercise}
