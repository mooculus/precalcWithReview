	In each part, an invertible function $f$ will be defined. For each function, find its inverse.

	\begin{exercise}	
		$f(x) = 5x + 3$
		
		\[f^{-1}(x) = \answer{\frac{x-3}{5}}\]
	\end{exercise}
	
	\begin{exercise}	
		$f(x) = \frac{x - 4 }{7} - 2$
		
		\[f^{-1}(x) = \answer{7(x+2)+ 4}\]
	\end{exercise}

	\begin{exercise}			
		$f(x) = \sqrt[3]{3-x} + 1$
		
		\[f^{-1}(x) = \answer{3-(x - 1)^3}\]
	\end{exercise}
