\begin{exercise}
Solve the following system of equations.\\
\[
\begin{cases}
-2(\sin(a))^2 + 2(\cos(b))^2 = -1\\
3(\sin(a))^2 - (\cos(b))^2 = 2
\end{cases}
\]
\[
a = \answer{\frac{\pi}{3}} b = \answer{\frac{\pi}{3}}
\]

\end{exercise}
