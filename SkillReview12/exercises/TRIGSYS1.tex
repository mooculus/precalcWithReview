\begin{exercise}
Solve the following system of equations for $x$ and $y$ between $0$ and $\frac{\pi}{2}$.\\
\[
\begin{cases}
2\sin(y) + \cos(x) = 1\\
4\sin(y) - \cos(x) = 2
\end{cases}
\]
\[
x = \answer{\frac{\pi}{2}} y = \answer{\frac{\pi}{6}}
\]

\end{exercise}
