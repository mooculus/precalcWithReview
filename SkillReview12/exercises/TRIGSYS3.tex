\begin{exercise}
Solve the following system of equations.  Assume $0 \leq a \leq \frac{\pi}{2}$ and $0 \leq b \leq \frac{\pi}{2}$.\\
\[
\begin{cases}
2(\tan(a))^2 + 3(\tan(b))^2 = 7\\
-(\tan(a))^2 + 6(\tan(b))^2 = -1
\end{cases}
\]
\[
a = \answer{\frac{\pi}{3}} b = \answer{\frac{\pi}{6}}
\]

\end{exercise}
