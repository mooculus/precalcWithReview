\begin{exercise}
Solve the following system of equations.\\
\[
\begin{cases}
2(\tan(a))^2 + 3(\tan(b))^2 = 7\\
-(\tan(a))^2 + 6(\tan(b))^2 = -1
\end{cases}
\]
\[
a = \answer{\frac{\pi}{3}} b = \answer{\frac{\pi}{6}}
\]

\end{exercise}
