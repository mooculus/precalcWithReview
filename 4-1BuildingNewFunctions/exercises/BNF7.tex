\begin{exercise}
Let $h$ be a function defined by $h(x) = 5\tan(x)$. Which of the following definitions of $f$ and $g$ satisfy $\left(\frac{f}{g}\right)(x) = h(x)$?
\begin{multipleChoice}
\choice{$f(x) = 5\sin(x)$ and $g(x) = 5\cos(x)$}
\choice{$f(x) = \sin(x)$ and $g(x) = \cos(x)$}
\choice[correct]{$f(x) = \sin(x)$ and $g(x) = \frac{\cos(x)}{5}$}
\choice{$f(x) = 5\sin(x)$ and $g(x) = \frac{1}{5\cos(x)}$}
\end{multipleChoice}

Let $h$ be a function defined by $h(x) = 2\tan(x)$. Which of the following definitions of $f$ and $g$ satisfy $(f \cdot g)(x) = h(x)$?
\begin{multipleChoice}
\choice{$f(x) = \tan(x)$ and $g(x) = 2\cos(x)$}
\choice[correct]{$f(x) = 2\sin(x)$ and $g(x) = \frac{1}{\cos(x)}$}
\choice{$f(x) = \cos(x)$ and $g(x) = 2\sin(x)$}
\choice{$f(x) = \frac{2}{\sin(x)}$ and $g(x) = \cos(x)$}
\end{multipleChoice}

Let $h$ be a function defined by $h(x) = \sin(x)\tan(x)$. Which of the following definitions of $f$ and $g$ satisfy $(f \cdot g)(x) = h(x)$?
\begin{multipleChoice}
\choice{$f(x) = (\sin(x))^2$ and $g(x) = \cos(x)$}
\choice{$f(x) = \sin(x)$ and $g(x) = \frac{1}{\cos(x)}$}
\choice[correct]{$f(x) = (\sin(x))^2$ and $g(x) = \frac{1}{\cos(x)}$}
\choice{$f(x) = \cos(x)$ and $g(x) = \left(\frac{1}{\sin(x)}\right)^2$}
\end{multipleChoice}

\end{exercise}
