\begin{exercise}
Consider the functions $f$ and $g$ defined by the table of values below. \\
		\begin{minipage}{.5\textwidth}
		  	\begin{center}
				$\begin{array}{||c|c||}
					\hline 
					\hline
					x & f(x)\\
					\hline
					1&3\\
					\hline
					2&2\\
					\hline
					3&2\\
					\hline
					4& 1\\
					\hline 
					\hline
				\end{array}$
			\end{center}
		\end{minipage}% This must go next to `\end{minipage}`
		\begin{minipage}{.5\textwidth}
		  	\begin{center}
				$\begin{array}{||c|c||}
					\hline 
					\hline
					x & g(x)\\
					\hline
					1&1\\
					\hline
					2&1\\
					\hline
					3&4\\
					\hline
					4& 6\\
					\hline 
					\hline
				\end{array}$
			\end{center}
		\end{minipage}
Use the pair of functions $f$ and $g$ to find the following values, if they exist. If the value does not exist, enter DNE.
\begin{enumerate}
\item $(f \cdot g)(3) = \answer{8}$
\item $(g \circ f)(2) = \answer{1}$
\item $(f + g)\left(1\right) = \answer{4}$
\item $(f\circ g)\left(3\right) = \answer{1}$
\item $(f + f)(1) = \answer{6}$
\item $(g - g)(3) = \answer{0}$
\end{enumerate}

\end{exercise}
