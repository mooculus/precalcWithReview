\begin{exercise}
Solve the equation for the given variable.\\
$\frac{3}{5}x+ \frac{1}{2}y=\frac{3}{7}$\\
$y=\answer{\frac{\frac{3}{7}-\frac{3}{5}x}{\frac{1}{2}}}$
$x=\answer{\frac{\frac{3}{7}-\frac{1}{2}y}{\frac{3}{5}}}$
\end{exercise}
