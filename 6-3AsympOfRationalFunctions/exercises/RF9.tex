\begin{exercise}


  Consider the rational function $$    f(x) = \frac{(x^4+1)^3}{(x^2-13x+36)^2}  $$

  \begin{enumerate}
  \item The values $c_1$ and $c_2$ for which $f(x)$ is undefined are, in increasing order, $c_1 = \answer{4}$ and $c_2 = \answer{9}$.
  \item For the value $c_1$, we have a
    \begin{multipleChoice}
      \choice{Hole in the graph of $y = f(x)$.}
      \choice[correct]{Vertical asymptote of line equation $y=c_1$.}
    \end{multipleChoice}
  \item For the value $c_2$, we have a
    \begin{multipleChoice}
      \choice{Hole in the graph of $y = f(x)$.}
      \choice[correct]{Vertical asymptote of line equation $y=c_2$.}
    \end{multipleChoice}
  \end{enumerate}

\end{exercise}
