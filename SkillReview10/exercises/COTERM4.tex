\begin{exercise}
Select all of the following angles that are coterminal to $\frac{2\pi}{5}$ radian.
  \begin{selectAll}  
    \choice{$\frac{-2\pi}{5}$}  
\choice{$\frac{4\pi}{5}$}  
\choice[correct]{$\frac{12\pi}{5}$}  
\choice{$\frac{5\pi}{5}$}  
\choice[correct]{$\frac{22\pi}{5}$}  
\choice[correct]{$\frac{-8\pi}{5}$}  

  \end{selectAll}
\end{exercise}
