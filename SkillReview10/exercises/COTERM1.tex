\begin{exercise}
Select all of the following angles that are coterminal to $1$ radian.
  \begin{selectAll}  
    \choice{$1+\pi$}  
    \choice[correct]{$1+2\pi$}  
    \choice{$1+3\pi$}  
    \choice[correct] {$1 + 4\pi$}
\choice{$1-\pi$}  
 \choice[correct]{$1-2\pi$}  
 \choice{$2\pi-1$}  
  \end{selectAll}
\end{exercise}
