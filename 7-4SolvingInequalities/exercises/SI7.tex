In this problem, we will review some properties of inequalities. For each statement, choose all the statements that logically follow. 
\begin{exercise}

\begin{enumerate}
\item For all $x$ in the interval $(0, 3)$, $f(x) < g(x) < h(x)$.
\begin{selectAll}
\choice[correct]{For all $x$ in the interval $(0, 3)$, $f(x) \le g(x)$.}
\choice{For all $x$ in the interval $(0, 3)$, $f(x) = g(x)$.}
\choice[correct]{For all $x$ in the interval $(0, 3)$, $h(x) > g(x)$.}
\choice[correct]{For all $x$ in the interval $(0, 3)$, $f(x) < h(x)$.}
\choice[correct]{For all $x$ in the interval $(0, 3)$, $f(x) \le h(x)$.}
\choice{For all $x$ in the interval $(0, 3)$, $h(x) \le g(x)$.}
\choice[correct]{For all $x$ in the interval $(0, 3)$, $h(x) \ge g(x)$.}
\end{selectAll}

\item For all $x$ in the interval $(-2, 5)$, $f(x) < h(x)$ and $g(x) = h(x)$.
\begin{selectAll}
\choice{For all $x$ in the interval $(-2, 5)$, $f(x) \ge g(x)$.}
\choice{For all $x$ in the interval $(-2, 5)$, $f(x) = g(x)$.}
\choice{For all $x$ in the interval $(-2, 5)$, $h(x) > g(x)$.}
\choice[correct]{For all $x$ in the interval $(-2, 5)$, $f(x) < g(x)$.}
\choice[correct]{For all $x$ in the interval $(-2, 5)$, $f(x) \le h(x)$.}
\choice[correct]{For all $x$ in the interval $(-2, 5)$, $h(x) \le g(x)$.}
\choice[correct]{For all $x$ in the interval $(-2, 5)$, $h(x) \ge g(x)$.}
\choice{For all $x$ in the interval $(-2, 5)$, $g(x) < f(x)$.}
\end{selectAll}

\item For all $x$ in the interval $(-\infty, 1)$, $f(x) \le g(x)$ and $f(x) = h(x)$.
\begin{selectAll}
\choice{For all $x$ in the interval $(-\infty, 1)$, $f(x) \ge g(x)$.}
\choice{For all $x$ in the interval $(-\infty, 1)$, $f(x) = g(x)$.}
\choice{For all $x$ in the interval $(-\infty, 1)$, $h(x) > g(x)$.}
\choice{For all $x$ in the interval $(-\infty, 1)$, $f(x) < g(x)$.}
\choice[correct]{For all $x$ in the interval $(-\infty, 1)$, $f(x) \le h(x)$.}
\choice[correct]{For all $x$ in the interval $(-\infty, 1)$, $h(x) \le g(x)$.}
\choice{For all $x$ in the interval $(-\infty, 1)$, $h(x) \ge g(x)$.}
\choice{For all $x$ in the interval $(-\infty, 1)$, $g(x) < f(x)$.}
\end{selectAll}
\end{enumerate}


\end{exercise}
