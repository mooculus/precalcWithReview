Solve the following inequalities.
\begin{exercise}
\begin{enumerate}
\item Solve the inequality $x^2 > x$.

\begin{multipleChoice}
\choice{$[0, 1]$}
\choice{$(0, 1)$}
\choice[correct]{$(-\infty, 0) \cup (1, \infty)$}
\choice{$(-\infty, 0] \cup [1, \infty)$}
\end{multipleChoice}

\item Solve the inequality $\frac{1}{x^2} \le \frac{1}{x}$.

\begin{multipleChoice}
\choice{$(-\infty, 0) \cup (1, \infty)$}
\choice{$(-\infty, 0) \cup [1, \infty)$}
\choice[correct]{$[1, \infty)$}
\choice{$(1, \infty)$}
\choice{$(0, 1)$}
\choice{$(0, 1]$}
\choice{$(-\infty, 0)$}
\choice{$(-\infty, 0) \cup (0, 1)$}
\choice{$(-\infty, 0) \cup (0, 1]$}
\end{multipleChoice}

\end{enumerate}

\end{exercise}
