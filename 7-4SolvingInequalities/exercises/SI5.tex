\begin{exercise}
\begin{enumerate}
\item Solve the inequality $-x^2 + 2 > x + 2$.

\begin{multipleChoice}
\choice{$[-1, 0]$}
\choice[correct]{$(-1, 0)$}
\choice{$(-\infty, -1) \cup (1, \infty)$}
\choice{$(-\infty, -1] \cup [1, \infty)$}
\end{multipleChoice}

\item Solve the inequality $-x^2 + 2 \le x + 2$.

\begin{multipleChoice}
\choice{$[-1, 0]$}
\choice{$(-1, 0)$}
\choice{$(-\infty, -1) \cup (1, \infty)$}
\choice[correct]{$(-\infty, -1] \cup [1, \infty)$}
\end{multipleChoice}

\end{enumerate}

\end{exercise}
