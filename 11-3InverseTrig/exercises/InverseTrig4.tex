\begin{exercise}
Solve the following equations. Give exact solutions or write ``none" if there is no solution. 
%
\begin{enumerate}

\item $\arccos(2x) = \pi.$\\
$x = \answer{-\frac{1}{2}}$

\item $\arcsin(x^3 - \frac{3}{8}) = -\frac{\pi}{6}$\\
$x = \answer{-\frac{1}{2}}$

\item $\arctan(4x) = \pi$\\
$x = \answer{none}$
\begin{exercise}
Why is there no solution?
\begin{multipleChoice}
\choice[correct]{$\pi$ is not in the range of arctangent}
\choice{$\pi$ is not in the domain of arctangent}
\choice{$4x>\frac{\pi}{2}$, so it is not in the domain of arctangent}
\choice{$4x>\frac{\pi}{2}$, so it is not in the range of arctangent}
\end{multipleChoice}
\end{exercise}

\item $6\arctan^2(x)=\pi \arctan(x) + \pi^2$\\
$x = \answer{-\sqrt{3}}$

\item $\arcsin(x^4+x^2+1) = -1$ \\
$x = \answer{none}$
\begin{exercise}
Why is there no solution?
\begin{multipleChoice}
\choice{$-1$ is not in the range of arcsine}
\choice[correct]{$x^4 + x^2 +1>1$, so it is not in the domain of arcsine}
\choice{$-1$, so it is not in the domain of arcsine}
\choice{$x^4 + x^2 >1$, so it is not in the range of arcsine}
\end{multipleChoice}
\end{exercise}

\item $\arccos(x^2-1) = \frac{\pi}{2}$ \\
$x = \answer{-1}$ (smaller)\\
$x = \answer{1}$ (larger)

\item $\arccos(x^2+3) = \pi$ \\
$x = \answer{none}$
\begin{exercise}
Why is there no solution?
\begin{multipleChoice}
\choice{$\pi$ is not in the range of arccosine}
\choice{$\pi$ is not in the domain of arccosine}
\choice{$x^2+3>1$, so it  is not in the range of arccosine}
\choice[correct]{$x^2+3>1$, so it is not in the domain of arccosine}
\end{multipleChoice}
\end{exercise}

\end{enumerate}

\end{exercise}
