\begin{exercise}
Choose the polynomial whose graph is shown in each part.

\begin{enumerate}
\item 
\begin{image}
\begin{tikzpicture}
    \begin{axis}[ymin=-7, ymax=7]
         \addplot[<->, domain=-7:7,samples=150]{(x - 1)*(x + 1)^2} node{};
	%\node(label) at (axis cs:-6,6) {};	
    \end{axis}
\end{tikzpicture}
\end{image} 

\begin{multipleChoice}
\choice[correct]{$(x - 1)(x + 1)^2$}
\choice{$(x - 1)(x + 1)$}
\choice{$(x - 1)^3$}
\choice{$-(x - 1)(x + 1)^2$}
\end{multipleChoice}

\item 
\begin{image}
\begin{tikzpicture}
    \begin{axis}[ymin=-7, ymax=7]
         \addplot[<->, domain=-7:7,samples=150]{(x + 2)*x} node{};
	%\node(label) at (axis cs:-6,6) {};	
    \end{axis}
\end{tikzpicture}
\end{image} 

\begin{multipleChoice}
\choice{$x^2 - 2x$}
\choice{$x + 2$}
\choice{$2x^2 + 2$}
\choice[correct]{$x^2 + 2x$}
\end{multipleChoice}

\item 
\begin{image}
\begin{tikzpicture}
    \begin{axis}[ymin=-7, ymax=7]
         \addplot[<->, domain=-5:5,samples=150]{-(x - 2)^2*(x + 1)^2} node{};
	%\node(label) at (axis cs:-6,6) {};	
    \end{axis}
\end{tikzpicture}
\end{image} 

\begin{multipleChoice}
\choice{$(x - 2)^2(x + 1)^2$}
\choice[correct]{$-(x - 2)^2(x + 1)^2$}
\choice{$-(x - 2)(x + 1)^2$}
\choice{$(x +2)^2(x - 1)$}
\end{multipleChoice}

\item 
\begin{image}
\begin{tikzpicture}
    \begin{axis}[ymin=-7, ymax=7]
         \addplot[<->, domain=-6:6,samples=150]{-(x + 4)*x^2} node{};
	%\node(label) at (axis cs:-6,6) {};	
    \end{axis}
\end{tikzpicture}
\end{image} 

\begin{multipleChoice}
\choice{$x^2 + 4x$}
\choice{$-x^2 - 4x$}
\choice{$x^3 - 4x^2$}
\choice[correct]{$-x^3 - 4x^2$}
\end{multipleChoice}

\item 
\begin{image}
\begin{tikzpicture}
    \begin{axis}[ymin=-7, ymax=7]
         \addplot[<->, domain=-6:7,samples=150]{(x - 6)*(x - 5)} node{};
	%\node(label) at (axis cs:-6,6) {};	
    \end{axis}
\end{tikzpicture}
\end{image} 

\begin{multipleChoice}
\choice{$(x - 6)(-x + 5)$}
\choice{$(x - 6)(-x - 5)$}
\choice[correct]{$(x - 6)(x - 5)$}
\choice{$(-x + 6)(x - 5)$}
\end{multipleChoice}

\end{enumerate}
\end{exercise}
