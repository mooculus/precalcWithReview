\begin{selectAll}
\choice{$f(x) = x^2$ }

\choice[correct]{$f(x) = x^2$ for $x \ge 0$}

\choice[correct]{$f(x) = x^2$ for $x \le 0$}
\choice[correct]{$f(x) = x^2$ for $x > 0$} 


\choice[correct]{$f(x) = x^2$ for $x \ge 3$}

\choice{$f(x) = x^2$ for $x \ge -1$}
\choice[correct]{$f(x) = x^2$ for $1 \le x \le 9$}
\end{selectAll}
