\begin{exercise}
Select all of the following statements below that are true. 

\begin{selectAll}
\choice[correct]{All linear functions with nonzero slope are invertible.}
\choice{All quadratic functions are invertible.}
\choice{All polynomials are invertible.}
\choice[correct]{All transformations (horizontal and vertical compressions, stretches, shifts, and reflections) of exponential functions are invertible.}
\choice[correct]{If the function $f$ is invertible, then so is $f^{-1}$.}
\choice{If the functions $f$ and $g$ are invertible, so is the function $f + g$.}
\choice{If the function $f$ is invertible, $f^{-1}(x) = -f(-x)$.}
\choice[correct]{If the point $(8, 12)$ is on the graph of an invertible function $f$, the point $(12, 8)$ is on the graph of $f^{-1}$.}
\choice[correct]{If the functions $f$ and $g$ are invertible, so is the function $f \circ g$.}
\end{selectAll}

\end{exercise}
